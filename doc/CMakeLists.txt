find_package(Doxygen)

option(Gear2D_INSTALL_DOC "Build and install documentation generated by doxygen" ON)

if(DOXYGEN_FOUND)
configure_file(${CMAKE_CURRENT_SOURCE_DIR}/Doxyfile.in ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile @ONLY)
add_custom_target(doc
${DOXYGEN_EXECUTABLE} ${CMAKE_CURRENT_BINARY_DIR}/Doxyfile
WORKING_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}
COMMENT "Generating API documentation with Doxygen" VERBATIM
)

add_custom_command(TARGET doc
	POST_BUILD
	COMMAND ${CMAKE_COMMAND} -E copy "${CMAKE_CURRENT_SOURCE_DIR}/navtree.css" "${CMAKE_CURRENT_BINARY_DIR}/html/navtree.css"
	COMMAND ${CMAKE_COMMAND} -E copy "${CMAKE_CURRENT_SOURCE_DIR}/resize.js" "${CMAKE_CURRENT_BINARY_DIR}/html/resize.js"
	COMMAND ${CMAKE_COMMAND} -E copy "${CMAKE_CURRENT_SOURCE_DIR}/search.css" "${CMAKE_CURRENT_BINARY_DIR}/html/search/search.css"
)


if (Gear2D_INSTALL_DOC)
add_dependencies(gear2d doc)
message(STATUS "Documentation will be generated and installed for Gear2D")
install(DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/html DESTINATION share/doc/${PROJECT_NAME})
endif(Gear2D_INSTALL_DOC)

endif(DOXYGEN_FOUND)
